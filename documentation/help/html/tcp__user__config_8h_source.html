15// =============================================================================
16// TCP Receive Accumulation Buffer
17// =============================================================================
18// Maximum number of bytes that can be accumulated while waiting for a complete
19// TCP/IP OpenLCB message (preamble + body). Must be large enough to hold the
20// largest expected single message. The spec allows multi-part messages, so
21// each part must fit within this buffer.
22//
23// Minimum practical value: 17 (preamble) + 2 (MTI) + 6 (source) = 25
24// Default 1024 is generous for most use cases.
25
26#define USER_DEFINED_TCP_RX_ACCUMULATION_BUFFER_LEN 1024

30// =============================================================================
31// Maximum size of the stack-allocated buffer used to build outgoing TCP/IP
32// OpenLCB messages (preamble + body). Must be at least TCP_PREAMBLE_LEN (17)
33// plus the largest payload you expect to send.
34
35#define USER_DEFINED_TCP_TX_BUFFER_LEN 1024
