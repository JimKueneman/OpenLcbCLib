43// General boot initialization
44#define RUNSTATE_INIT 0
45// Generate the 48 bit seed to create the Alias from
46#define RUNSTATE_GENERATE_SEED 1
47// Generate the Alias
48#define RUNSTATE_GENERATE_ALIAS 2
49// Send the 4 (CID) messages with the NodeID and suggested Alias

51#define RUNSTATE_LOAD_CHECK_ID_06 4
52#define RUNSTATE_LOAD_CHECK_ID_05 5
53#define RUNSTATE_LOAD_CHECK_ID_04 6
54// Wait for 200ms to see if anyone objects. They objection could occur in this or the previous state,
55// if they do then jump back to RUNSTATE_GENERATE_SEED to try again
56#define RUNSTATE_WAIT_200ms 7
57// Send the Alias Reserved message (AMR)
58#define RUNSTATE_LOAD_RESERVE_ID 8
59// At this point the alias is reserved
60// To "log in" send the (AMD) message, this sets the node to "Permitted" but still not OpenLCB "Initialized"
61#define RUNSTATE_LOAD_ALIAS_MAP_DEFINITION 9
62// At this point the CAN specific login is complete, the Alias is reserved and the node is ready to be an OpenLCB Node
63#define RUNSTATE_LOAD_INITIALIZATION_COMPLETE 10
64// Node is Initialized and can send any message, need to send the events we handle
65#define RUNSTATE_LOAD_CONSUMER_EVENTS 11
66// Node is Initialized and can send any message, need to send the events we handle
67#define RUNSTATE_LOAD_PRODUCER_EVENTS 12
68// Runs the message loop
69#define RUNSTATE_RUN 13
